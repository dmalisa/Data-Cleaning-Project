# c.
# remove all the numbers from the address as well. 
import re
def get_street_type(address):
    pattern = r'[0-9]'
    line = re.sub(pattern, "",address)
    if "Unit" in address:
        i = line.find("Unit")
        before = line[:i]
        street_type = list(filter(None,before.split(" ")))[-1]
    else:
        street_type = list(filter(None,line.split(" ")))[-1]
    return street_type
